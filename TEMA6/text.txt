int main ()
{
   int a , maxim ;
   a = 12345 , maxim = 0 ;
   while ( 0 < a)
   {
       if ( maxim < a % 10 )
       maxim = a % 10 ;
       a = a / 10 ;
   }
    for ( int i = 0 ; i < 5 ; i = i + 1 )
   {
      if ( 0 < i + maxim )
           maxim = maxim + i ;
     else
          maxim = maxim - i;

          
   }

}
